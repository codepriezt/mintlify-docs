import { useState } from 'react';

export const Feedback = ({ pageUrl }) => {
  const [feedback, setFeedback] = useState('');
  const [rating, setRating] = useState(null);
  const [submitted, setSubmitted] = useState(false);
  const [error, setError] = useState(null);

  const handleSubmit = async (e) => {
    e.preventDefault();
    
    try {
      const response = await fetch('YOUR_SHEETY_API_ENDPOINT', {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
        },
        body: JSON.stringify({
          feedback: {
            pageUrl,
            rating,
            feedback,
            timestamp: new Date().toISOString()
          }
        })
      });

      if (!response.ok) throw new Error('Failed to submit feedback');
      
      setSubmitted(true);
      setFeedback('');
      setRating(null);
      setError(null);
    } catch (err) {
      setError('Failed to submit feedback. Please try again.');
    }
  };

  if (submitted) {
    return (
      <div className="mt-8 p-4 bg-green-50 dark:bg-green-900 rounded-lg">
        <p className="text-green-700 dark:text-green-100">
          Thank you for your feedback! We appreciate your help in improving our documentation.
        </p>
      </div>
    );
  }

  return (
    <div className="mt-8 border-t pt-8">
      <h3 className="text-xl font-semibold mb-4">Was this page helpful?</h3>
      
      <div className="flex gap-4 mb-4">
        {[1, 2, 3, 4, 5].map((value) => (
          <button
            key={value}
            onClick={() => setRating(value)}
            className={`p-2 rounded-full w-10 h-10 ${
              rating === value
                ? 'bg-primary text-white'
                : 'bg-gray-100 dark:bg-gray-800'
            }`}
          >
            {value}
          </button>
        ))}
      </div>

      <form onSubmit={handleSubmit} className="space-y-4">
        <div>
          <label htmlFor="feedback" className="block mb-2">
            Additional feedback (optional)
          </label>
          <textarea
            id="feedback"
            value={feedback}
            onChange={(e) => setFeedback(e.target.value)}
            className="w-full p-2 border rounded-lg dark:bg-gray-800"
            rows="3"
            placeholder="Tell us what we can improve..."
          />
        </div>

        {error && (
          <p className="text-red-600 dark:text-red-400">{error}</p>
        )}

        <button
          type="submit"
          disabled={!rating}
          className={`px-4 py-2 rounded-lg ${
            rating
              ? 'bg-primary text-white hover:bg-primary-dark'
              : 'bg-gray-100 text-gray-400 cursor-not-allowed dark:bg-gray-800'
          }`}
        >
          Submit Feedback
        </button>
      </form>
    </div>
  );
};
